const socket = io('http://localhost:5000');

socket.on('connect', function() {
    console.log('Connected to the server');
});

socket.on('battery_update', function(data) {
    console.log('Battery level:', data.battery_level);
    // Update the UI with the new battery level
});

socket.on('disconnect', function() {
    console.log('Disconnected from the server');
});


// Assuming socket is already connected to your Flask-SocketIO server
socket.emit('drone_command', {command: 'w'}); // Command to move the drone forward



socket.on('video_frame', function(data) {
    var videoFeed = document.getElementById('video_feed');
    videoFeed.src = 'data:image/jpeg;base64,' + data.image;
});
